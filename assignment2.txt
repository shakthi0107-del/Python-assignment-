# Create a dictionary for one department with keys "budget" and "employees"

           ]

   }
# Calculate and print the total salary, then print if it is within the budget

# Your code here:
department = {
    "Engineering": {
        "budget": 500000,
        "employees": [
            {"name": "sakthi", "salary": 90000},
            {"name": "Abi",   "salary": 85000}
        ]
    }
}


E = department["Engineering"]

total_salary = E["employees"][0]["salary"] + E["employees"][1]["salary"]

print("Total Salary:", total_salary)

if total_salary <= E ["budget"]:
    print("The total salary is within the budget.")
else:
    print("The total salary exceeds the budget.")

# Exercise 4:
# Create two sets: tech_subscribers and news_subscribers with some overlapping emails
# Print the emails that are in both sets
# Print all unique emails from both sets
# Remove one email from both sets and print the updated sets

# Your code here:
L2 = {"a@gmail.com", "b@gmail.com", "c@gmail.com"}
L3 = {"b@gmail.com", "c@gmail.com", "d@gmail.com"}


common_emails = L2.intersection(L3)
print("Common emails:", common_emails)


all_unique_emails = L2.union(L3)
print("All unique emails:", all_unique_emails)


L2.remove("a@gmail.com")
L3.remove("d@gmail.com")


print("Updated L2:", L2)
print("Updated L3:", L3)

# Exercise 5: 
# Create a dictionary "library" with 3 books; each value is a dictionary with "title", "author", "available"
# Create a list "borrowing_history" with 2 tuples (book_id, user_name)
# Print all titles of books that are available
# Print all user names that appear in borrowing_history

# Your code here:

library = {
    "book1": {"title": "The Alchemist", "author": "Paulo Coelho", "available": True},
    "book2": {"title": "1984", "author": "George Orwell", "available": False},
    "book3": {"title": "Atomic Habits", "author": "James Clear", "available": True}
}


borrowing_history = [
    ("book1", "Sakthi"),
    ("book2", "Abi")
]


print("Available books:")
for book_id, details in library.items():
    if details["available"]:
        print("-", details["title"])


print("\nUsers who borrowed books:")
for entry in borrowing_history:
    user_name = entry[1]
    print("-", user_name)
